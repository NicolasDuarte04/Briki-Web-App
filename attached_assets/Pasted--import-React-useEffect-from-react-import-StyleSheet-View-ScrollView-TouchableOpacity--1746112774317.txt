            Compare and choose the best travel insurance for your journey
          </ThemedText>
          <Animated.View style={pulseStyle}>
            <TouchableOpacity style={styles.ctaButton} onPress={handleGetQuote}>
              <ThemedText style={styles.ctaButtonText}>Get Insured Now</ThemedText>
            </TouchableOpacity>
          </Animated.View>
        </Animated.View>

        <Animated.View entering={FadeInUp.delay(300)} style={styles.aboutSection}>
          <ThemedText style={styles.sectionTitle}>About Briki</ThemedText>

            <Animated.View entering={FadeInUp.delay(400).springify()} style={styles.aboutCard}>
