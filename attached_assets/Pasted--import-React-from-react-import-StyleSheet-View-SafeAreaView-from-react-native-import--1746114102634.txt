
import React from 'react';
import { StyleSheet, View, SafeAreaView } from 'react-native';
import { router, useLocalSearchParams } from 'expo-router';
import { ThemedText } from '@/components/ThemedText';
import { TouchableOpacity } from 'react-native-gesture-handler';
import { colors, fonts, buttonStyles, shadows } from './constants/Styles';
import { Ionicons } from '@expo/vector-icons';

export default function PaymentSuccessScreen() {
  const params = useLocalSearchParams();
  const { plan, amount, reference } = params;
  const planData = plan ? JSON.parse(decodeURIComponent(plan as string)) : null;

  const handleDone = () => {
    router.replace('/(tabs)');
  };

  return (
    <SafeAreaView style={styles.container}>
      <View style={styles.content}>
        <Ionicons 
          name="checkmark-circle" 
          size={64} 
          color="#4BB543" 
          style={styles.icon} 
        />
        <ThemedText style={styles.title}>Payment Successful!</ThemedText>
        <ThemedText style={styles.message}>
          Your travel insurance has been successfully purchased. You'll receive a confirmation email shortly.
        </ThemedText>

        <View style={styles.detailsContainer}>
          <View style={styles.detailRow}>
            <ThemedText style={styles.detailLabel}>Insurance Provider</ThemedText>
            <ThemedText style={styles.detailValue}>{planData?.provider}</ThemedText>
          </View>
          <View style={styles.detailRow}>
            <ThemedText style={styles.detailLabel}>Amount Paid</ThemedText>
            <ThemedText style={styles.detailValue}>${amount}</ThemedText>
          </View>
          <View style={styles.detailRow}>
            <ThemedText style={styles.detailLabel}>Reference No.</ThemedText>
            <ThemedText style={styles.detailValue}>{reference}</ThemedText>
          </View>
        </View>

        <TouchableOpacity style={styles.button} onPress={handleDone}>
          <ThemedText style={styles.buttonText}>Back to Home</ThemedText>
        </TouchableOpacity>
      </View>
    </SafeAreaView>
  );
}

  container: {
    flex: 1,
    backgroundColor: colors.background,
  },
  content: {
    flex: 1,

  },

    fontSize: 28,

  },
  message: {
    fontSize: 16,

    textAlign: 'center',
    marginTop: 8,
  },
  button: {
    ...buttonStyles.primary,
    minWidth: 200,
    marginTop: 32,
  },
  buttonText: buttonStyles.primaryText,
  icon: {
    marginBottom: 24,
  },
  detailsContainer: {
    backgroundColor: colors.background,
    borderRadius: 16,
    padding: 24,
